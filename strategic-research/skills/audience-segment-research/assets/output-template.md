# audience-segment-research handoff
version: 1
generated_at: {ISO-8601 date}

framing:
  anchor: {name or null}
  industry: {name}
  b2b_or_b2c: {b2b | b2c | hybrid}
  lifecycle_stage: {pre-pmf | early-pmf | scaling | mature}
  end_user_persona: {persona}
  scope_boundary: {sentence}
  upstream_handoff_consumed: {true | false}

segmentation:
  primary_lens: {lens name}
  secondary_lens: {lens name or null}
  rationale: {one-paragraph rationale}
  scores:
    identifiability: {1-5}
    care_intensity: {1-5}
    behavioral_grounding: {1-5}
    downstream_actionability: {1-5}

segments:
  - id: {slug}
    name: {name}
    one_line: {one-sentence description}
    identifier:
      behavioral: [{action 1}, {action 2}]
      firmographic: [{descriptor 1}]
    prevalence_estimate: {percent}
    prevalence_confidence: {0-100}
    who_they_are: {paragraph}
    what_they_do: {verbs}
    jobs: [{JTBD 1}, {JTBD 2}]
    pains_unmet_needs:
      - pain: {description}
        source: {citation}
        evidence_type: {behavioral | stated}
    current_alternatives: [{alt 1}, {alt 2}]
    buying_triggers: [{trigger 1}]
    stakeholders:
      user: {title + role}
      champion: {title}
      economic_buyer: {title}
      approvers: [{role 1}]
      blocker: {role}
      renewer: {role}
      implementer: {role or null}
    tensions:
      - id: T-{NN}
        claim_a: {statement}
        claim_b: {statement}
        class: {subsegment-split | context-dependent | stated-vs-revealed | temporal | unresolved}
        resolution: {integration}
        resolution_confidence: {0-100}
    outlier_signals:
      - id: S-{NN}
        observation: {statement}
        source: {citation}
        hypothesis: {why it matters}
        promotes_to_pattern_when: {evidence threshold}
    confidence: {0-100}
    evidence_rollup:
      behavioral_observations: {n}
      stated_observations: {n}

cross_segment_tensions:
  - id: T-{NN}
    scope: {e.g., "all segments"}
    claim_a: {statement}
    claim_b: {statement}
    class: {class}
    resolution: {integration}
    resolution_confidence: {0-100}

global_outliers:
  - id: S-{NN}
    observation: {statement}
    source: {citation}
    hypothesis: {why it matters}
    promotes_to_pattern_when: {threshold}

cross_segment:
  shared_jobs: [{job 1}]
  shared_pains: [{pain 1}]
  divergent_jobs:
    - job: {statement}
      specific_to_segments: [{seg id 1}]
  adjacency:
    compatible_pairs: [[{seg1}, {seg2}]]
    cannibalization_risks: [[{seg1}, {seg2}]]

glossary:
  - term: {term}
    definition: {definition}

evidence_quality:
  total_observations: {n}
  behavioral_ratio: {0.0-1.0}
  primary_sources_distinct: {n}
  overall_confidence: {0-100}
  known_gaps: [{gap 1}]

open_questions:
  - question: {question}
    why_it_matters: {reason}
    suggested_method: {method}

rejected_lenses:
  - lens: {name}
    score_total: {sum}
    why_rejected: {reason}
